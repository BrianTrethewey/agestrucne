131  def parseNoDatFile( self ):
132  '''
133  This def is provisional, since I have only seen
134  a few of the *NoDat.txt files generated by
135  NeEstimator (v2). I base the parsing on examples like this:
136 
137  Population 1 [OmyLGRA12S_0213]
138  -----------------------------------------------------------
139  Individual Locus Genotype Number of Loci
140  with missing data
141  1 9 0000 4
142  ----------------------------------------------------------
143 
144 
145 
146  However thin the ground, this def assumes
147 
148  1. multipop results will simply have more table entries
149  like the one above.
150 
151  2. If a line in the NeEstimator's "*NoDat.txt" file
152  starts with the value below in COLHEADSTART, and
153  contains the other column keywords given below
154  in COLSINCLUDE, that is must be the header and
155  also that the data follows after one mostly line blank
156  but for a tag on the last column header
157 
158  3. Population name entry for each table is uniqely
159  located by the line starting with value below in POPLINESTART
160  '''
161 
162 
163  POPLINESTART="Population"
164  COLHEADSTART="Individual"
165  COLSINCLUDE=[ "Locus", "Genotype", "Number of Loci" ]
166  ENDTABLELINESTART="----------------------"
167 
168  s_name=self.__get_name_nodat_file()
169 
170  if s_name is None:
171  return
172  #end return if no nodat file
173 
174  self.__parsed_nodat_output=[]
175 
176  o_file=open( s_name )
177 
178  b_found_header=False
179  i_table_lines_count=0
180  s_currpop=""
181  for s_line in o_file:
182 
183  i_num_match=0
184  if s_line.startswith( POPLINESTART ):
185  #we replace spaces in the pop name:
186  s_currpop=s_line.strip().replace( " ", "_" )
187  elif s_line.startswith(COLHEADSTART):
188  lb_trues=[ s_col in s_line for s_col in COLSINCLUDE ]
189  if sum( lb_trues ) == len( COLSINCLUDE ):
190  b_found_header=True
191  i_table_lines_count=0
192  #end if all header names in line
193  else:
194  #first line after header is more header:
195  if b_found_header and i_table_lines_count > 1:
196  #if end of table for currpop,
197  #reset flag, line counts
198  if s_line.startswith( ENDTABLELINESTART ):
199  b_found_header=False
200  i_table_lines_count=0
201  else:
202  i_table_lines_count+=1
203  #NoDat.txt files data cols
204  #are seperated by multiple space chars,
205  #so we use the default split to
206  #list only non-space as items:
207  ls_vals=s_line.strip().split()
208  #first col is the pop name
209  self.__parsed_nodat_output.append( [ s_currpop ] \
210  + [ s_val for s_val in ls_vals ] )
211  #end if end-table line, else data
212  #end if header's been found and we're past the non-date line
213  #end if pop line else header line, else some other line
214  #end for each line in nodat file
215 
216  o_file.close()
