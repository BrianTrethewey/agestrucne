320 def get_genepop_file_list( s_genepop_files_arg ):
321  '''
322  if invoked from the console the arg is a glob expression
323  if invoked from the gui interface in pgguineestimator.py,
324  the arg is a list, such that eval will genreate a python
325  list of strings.
326 
327  This def assumes that no user will invoke this mod from
328  the command line with a glob expression, includeing a file name,
329  that can be evaluated as a python list (i.e. "[ \"item\", \"item\" ... ]
330 
331  '''
332 
333  ls_files=None
334 
335  try:
336 
337  ls_files=eval( s_genepop_files_arg )
338 
339  if type( ls_files ) != list:
340  #still could be a glob expression, for example,
341  #of a file name that python can eval as an int or float,
342  #so we'll just glob the orig:
343 
344  ls_files=glob.glob( s_genepop_files_arg )
345  else:
346  #make sure it's a list of strings
347  if not( file_name_list_is_all_strings( ls_files ) ):
348  s_msg="In pgdriveneestimator.py, def get_genepop_file_list, " \
349  + "genepop file arg evaluated as a list, but, not composed " \
350  + "of strings: " + str( list ) + "."
351  raise Exception( s_msg )
352  #end if not all strings in list
353  except ( NameError, SyntaxError ) as evalfail:
354 
355  '''
356  this is caught if the eval fails due to
357  the arg being a string. Further, a syntax
358  error results if the arg begins/ends in a wildcard
359  "*" char -- as will happen often with a glob.
360 
361  So for these eval failures we assume the arg
362  is a glob expression
363  '''
364  ls_files=glob.glob( s_genepop_files_arg )
