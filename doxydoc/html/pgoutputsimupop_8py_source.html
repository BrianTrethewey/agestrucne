152  def bz2CompressAllFiles(self):
153  '''
154  used code and advice in,
155  http://stackoverflow.com/questions/9518705/big-file-compression-with-python
156 
157  Note: checked the shutil documentation at https://docs.python.org/2/library/shutil.html
158  which warns of loss of meta file info (owner/group ACLs) when using shutil.copy() or shutil.copy2(). Unclear
159  whether this applies to the copyfileobj, though my few tests showd all of these were retained.
160  '''
161  for s_myfile in [ self.__outname, self.__errname, self.__megadbname, self.__confname ]:
162  if not self.__file_exists( s_myfile ):
163  self.__raise_file_not_found_error( s_myfile, "compress file with bz2" )
164  else:
165  with open( s_myfile, 'rb' ) as o_input:
166  with bz2.BZ2File( s_myfile + '.bz2', 'wb', compresslevel=9 ) as o_output:
167  #try/except overkill, but want logic to show
168  #we don't remove the input file
169  #if the copy op fails:
170  try:
171  shutil.copyfileobj( o_input, o_output )
172  except Exception, e:
173  raise e
174  #end try except
175  os.remove( s_myfile )
176  #end with bzfile for writing
177  #end with current file for reading
178  #end if file absent, else exists
179  #end for each file

183  def gen2Genepop( self, idx_allele_start, idx_allele_stop,
184  b_do_compress=True,
185  b_pop_per_gen=False ):
186 
187  '''
188  reads the *.gen file from the simuPop output
189  and produces a genepop file based on the gen file
190 
191  param idx_allele_start: int, one-based index giving the
192  item (loci) number of the first allele entry in the line of gen-file input
193  Example, if the *gen file has 10 microsates and 40 SNPs, then its first
194  10 allele entries (that follow the indiv number and the gneration number)
195  will be the microsats, and the final 40 allele entries will be the SNPs
196  If we wanted only the Microsats to be included in the gen file, then
197  this param would be "1", and the idx_allele_stop value would be 10.
198  If we want both msats and snps, we'd give 1 and 50 for these params.
199  If we want only SNPs, wed enter 11 for start allele and 50 for stop.
200 
201  param idx_allele_stop: int, one-based index giving the
202  item (loci) number of the last allele entry in the line of gen-file input
203 
204  param optional b_do_compress, if true, will bzip2 the output genepop file
205 
206  param optional b_pop_per_gen, if true, will insert a "Pop" line between the
207  generations, as given by the 2nd field in the gen file
208  '''
209 
210  o_genfile=None
211  o_genepopfile=None
212  s_genepop_filename=self.__genfile + "." + GENEPOP_FILE_EXTENSION
213  i_num_loci=(idx_allele_stop-idx_allele_start) + 1
214 
215  IDX_GEN_INDIV=0
216  IDX_GEN_GENERATION=1
217  IDX_FIRST_ALLELE=2
218 
219  i_genepop_already_exists=self.__file_exists( s_genepop_filename )
220 
221  if i_genepop_already_exists != FILE_DOES_NOT_EXIST:
222 
223  self.__raise_file_exists_error( s_genepop_filename,
224  "create genepop file from gen file" )
225  #end if genepop file already exists
226 
227  i_genfile_exists=self.__file_exists( self.__genfile )
228 
229  #if we have the uncompressed version available
230  #we use it:
231  if i_genfile_exists == FILE_EXISTS_UNCOMPRESSED \
232  or i_genfile_exists == FILE_EXISTS_AS_BOTH_UNCOMPRESSED_AND_BZ2:
233 
234  o_genfile=open( self.__genfile )
235 
236  elif i_genfile_exists == FILE_EXISTS_AS_BZ2:
237 
238  o_genfile=bz2.BZ2File( self.__genfile + ".bz2" )
239 
240  else:
241  self.__raise_file_not_found_error( self.__genfile, "convert gen file to genepop" )
242  #end if uncompressed only or uncomp. and compressed, else compressed only, else no file
243 
244  if b_do_compress == True:
245  o_genepopfile=bz2.BZ2File( s_genepop_filename + '.bz2', 'wb', compresslevel=9 )
246  else:
247  o_genepopfile=open( s_genepop_filename, 'w' )
248  #end if compress else don't
249 
250  #write header and loci list:
251 
252  o_genepopfile.write( "genepop from simupop run data file " \
253  + self.__genfile + "\n" )
254 
255  for i_locus in range( i_num_loci ):
256  o_genepopfile.write("l" + str(i_locus) + "\n" )
257  #end for each loci number
258 
259  #write indiv id and alleles, line by line:
260  i_currgen=None
261 
262  for s_line in o_genfile:
263 
264  ls_line = s_line.rstrip().split(" ")
265 
266  i_id = int(float(ls_line[IDX_GEN_INDIV]))
267  i_gen = int(float(ls_line[IDX_GEN_GENERATION]))
268  idx_first_allele_in_line=( idx_allele_start + IDX_FIRST_ALLELE ) - 1
269  idx_last_allele_in_line=( idx_allele_stop + IDX_FIRST_ALLELE ) - 1
270  ls_alleles = ls_line[ idx_first_allele_in_line : idx_last_allele_in_line + 1 ]
271 
272  if i_currgen is None or ( b_pop_per_gen == True and i_currgen != i_gen ):
273  o_genepopfile.write( "pop\n" )
274  i_currgen=i_gen
275  #end if new generation (== new population )
276  #end if we should make a new population for each generation
277 
278  o_genepopfile.write( str( i_id ) + "," + " ".join( ls_alleles ) + "\n" )
279 
280  #end for each line in gen file
281 
282  o_genfile.close()
283  o_genepopfile.close()
284 
285  return
