400  def __compute_n0_from_eff_size_info( self ):
401  '''
402  2016_11_04
403  Revising by incorporating Brian Trethway's new calc method
404  (adapted from his module Sample_testing.py, which he pushed to
405  our github repo a few days ago).
406  '''
407 
408  i_n0=None
409 
410  #First make sure we have the parmeters in our input object:
411  ls_required_params=[ "NbNc", "_PGInputSimuPop__Nb_from_eff_size_info", "survivalFemale",
412  "survivalMale", "maleProb" ]
413 
414  ls_missing_params=[]
415 
416  for s_param_name in ls_required_params:
417  if not hasattr( self, s_param_name ):
418  ls_missing_params.append( s_param_name )
419  #end if our input object does not have the attribute
420  #end for each param
421 
422  if len( ls_missing_params ) != 0:
423  s_msg="In PGInputSimuPop instance, def __compute_n0_from_eff_size_info, " \
424  + "Unable to caluclate N0 due to missing parameter(s): " \
425  + ", ".join( ls_missing_params ) + "."
426  raise Exception( s_msg )
427  #end if one or more params missing
428 
429  f_female_ratio = 1-self.maleProb
430  f_Nc = self.__Nb_from_eff_size_info / self.NbNc
431 
432  f_current_male_prop=self.maleProb
433  f_current_female_prop=f_female_ratio
434  f_cum_pop_porp = 1
435 
436  #Assumes male and female survivals have same length
437  for i_age in range(len(self.survivalMale)):
438  #calcualte new male Ratio
439  f_current_male_prop = f_current_male_prop * self.survivalMale[i_age]
440  #calculate new female ratio
441  f_current_female_prop = f_current_female_prop * self.survivalFemale[i_age]
442  #add to cumulative
443  f_cum_pop_porp+=f_current_male_prop
444  f_cum_pop_porp+=f_current_female_prop
445  #end for each age in male survival list
446 
447  #calulate N0
448  i_n0 = int( round ( f_Nc/f_cum_pop_porp ) )
449 
450  return i_n0
451  #end __compute_n0_from_eff_size_info
