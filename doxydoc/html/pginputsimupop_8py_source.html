331  '''

334  to find the attribute names for the parameters used in the simupop

336  '''

338  if self.__param_names is None:

344  o_parser=ConfigParser()
345  o_parser.optionxform=str
346 
347  ls_attribute_names=self.param_names.shortnames
348 
349  for s_attribute in ls_attribute_names:
350  if hasattr( self, s_attribute ):
351  #if this attribute was not in the original config file,
352  #but was set in the GUI, we rely on the param_names
353  #object (listing attribute names, their corresponding lable text,
354  #and their section names (as part of the parma_names.tags), to provide
355  #the section name, with which we update this input object:
356  if s_attribute not in self.__config_file_section_name_by_attribute_name:
357  s_new_entry_section=self.param_names.getConfigSectionNameForParam( s_attribute )
