133  def __reset_value( self, idx_val ):

137  v_currval=self.__value[ idx_val ]
138 
139  v_newval=None
140 
141  try:

143  #python accepts bool("mystring" ) as True,
144  #but we want 0 or 1 only in those fields:
145  if o_type==bool:
146  if int( o_entryval.get() ) not in [ 0, 1 ]:

148  raise ValueError( "Boolean type requires int 0 or 1" )
149  else:

163  self.__item_types[ idx_val ] = self.__type_replaces_none

169  elif o_entryval.get() == "None" and idx_val in self.__idx_none_values:
170  v_newval=None

173  else:
174  v_newval=o_type( o_entryval.get() )

177  except ValueError as ve:
